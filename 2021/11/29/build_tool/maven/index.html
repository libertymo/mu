maven插件使用

###1.整体配置

<?xml version="1.0" encoding="UTF-8"?>

<project xmlns="http://maven.apache.org/POM/4.0.0" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
  xsi:schemaLocation="http://maven.apache.org/POM/4.0.0 http://maven.apache.org/xsd/maven-4.0.0.xsd">
  <modelVersion>4.0.0</modelVersion>

  <groupId>org.example</groupId>
  <artifactId>maven_test</artifactId>
  <version>1.0-SNAPSHOT</version>

  <name>maven_test</name>

  <properties>
    <!-- pluginManagement全局配置 -->
    <project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>
    <maven.compiler.source>1.8</maven.compiler.source>
    <maven.compiler.target>1.8</maven.compiler.target>
  </properties>

  <dependencies>
    <dependency>
      <groupId>junit</groupId>
      <artifactId>junit</artifactId>
      <version>4.11</version>
      <scope>test</scope>
    </dependency>
    <dependency>
      <groupId>org.apache.commons</groupId>
      <artifactId>commons-collections4</artifactId>
      <version>4.2</version>
    </dependency>
  </dependencies>
  
  <build>
    <pluginManagement>
      <plugins>
        <!-- 清除编译文件 -->
        <plugin>
          <artifactId>maven-clean-plugin</artifactId>
          <version>3.1.0</version>
        </plugin>
        <!-- maven中默认只能读取resources文件夹下的资源，maven-resources-plugin可指定路径下文件，和resources、testResources配合使用-->
        <plugin>
          <artifactId>maven-resources-plugin</artifactId>
          <version>3.0.2</version>
        </plugin>
        <!-- 编译插件 -->
        <plugin>
          <artifactId>maven-compiler-plugin</artifactId>
          <version>3.8.0</version>
          <configuration>
            <skip>true</skip>
          </configuration>
        </plugin>
        <!-- mvn test -->
        <plugin>
          <artifactId>maven-surefire-plugin</artifactId>
          <version>2.22.1</version>
        </plugin>
        <!-- mvn package -->
        <plugin>
          <artifactId>maven-jar-plugin</artifactId>
          <version>3.0.2</version>
          <configuration>
            <archive>
              <manifest>
                <addClasspath>true</addClasspath>
                <!-- 指定三方包依赖路径 -->
                <classpathPrefix>lib/</classpathPrefix>
                <!-- 可执行文件要在manifest指定主类-->
                <mainClass>org.example.App</mainClass>
              </manifest>
            </archive>
            <excludes>
              <exclude>**/bin/*</exclude>
            </excludes>
          </configuration>
        </plugin>
        
        <plugin>
          <artifactId>maven-assembly-plugin</artifactId>
          <version>3.0.0</version>
        </plugin>
        
        <!-- 打包到本地仓库 -->
        <plugin>
          <artifactId>maven-install-plugin</artifactId>
          <version>2.5.2</version>
        </plugin>
        <!-- 打包到远程仓库 -->
        <plugin>
          <artifactId>maven-deploy-plugin</artifactId>
          <version>2.8.2</version>
        </plugin>
      </plugins>
    </pluginManagement>
   <!-- maven-assembly-plugin在pluginManagement不生效 -->
    <plugins>
      <plugin>
        <artifactId>maven-assembly-plugin</artifactId>
        <configuration>
          <descriptors>
            <descriptor>assembly.xml</descriptor>
          </descriptors>
        </configuration>
        <executions>
          <execution>
            <id>make-assembly</id>
            <!-- 和mvn package绑定，即执行package命令会执行assembly -->
            <phase>package</phase>
            <goals>
              <goal>single</goal>
            </goals>
          </execution>
        </executions>
      </plugin>
    </plugins>
    
    <!-- 如指定路径，则不使用默认路径 -->
<!--    <resources>-->
<!--      <resource>-->
<!--        <directory>src/main/resources</directory>-->
<!--        <includes>-->
<!--          <include>**/*.properties</include>-->
<!--        </includes>-->
<!--        <excludes>-->
<!--          <exclude>**/*.gradle</exclude>-->
<!--        </excludes>-->
<!--      </resource>-->
<!--    </resources>-->
<!--    <testResources>-->
<!--      <testResource>-->
<!--        <directory>aaa</directory>-->
<!--      </testResource>-->
<!--    </testResources>-->
  </build>
</project>

###2. maven-assembly-plugin配置文件assembly.xml

<?xml version='1.0' encoding='UTF-8'?>
<assembly xmlns="http://maven.apache.org/ASSEMBLY/2.1.0" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
          xsi:schemaLocation="http://maven.apache.org/ASSEMBLY/2.1.0 http://maven.apache.org/xsd/assembly-2.1.0.xsd">
    <id>demo</id>
    <formats>
    	<!-- 打包成tar.gz格式 -->
        <format>tar.gz</format>
    </formats>
    <includeBaseDirectory>false</includeBaseDirectory>
    <fileSets>
        <fileSet>
        		<!-- 打包配置文件到bin目录 -->
            <directory>target/classes/bin</directory>
            <outputDirectory>bin</outputDirectory>
            <includes>
                <include>**/*</include>
            </includes>
            <fileMode>0755</fileMode>
        </fileSet>
        <fileSet>
        	  <!-- 打包可执行jar到包根目录 -->
            <directory>target</directory>
            <outputDirectory>/</outputDirectory>
            <includes>
                <include>**/*.jar</include>
            </includes>
            <fileMode>0755</fileMode>
        </fileSet>

    </fileSets>
    <!-- 依赖的三方包处理，打到lib包-->
    <dependencySets>
        <dependencySet>
            <useProjectArtifact>true</useProjectArtifact>
            <outputDirectory>lib</outputDirectory>
            <scope>runtime</scope>
        </dependencySet>
    </dependencySets>
</assembly>
