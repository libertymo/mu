ant使用

  • 1.下载ant

    • 官网下载:http://ant.apache.org/
    • 一般放在/usr/local目录

  • 2.配置ant环境变量

    • 在~/.bash_profile或者~/.bashrc配置环境变量
    #ant
export ANT_HOME=/usr/local/apache-ant-1.10.12
export PATH=${PATH}:${ANT_HOME}/bin

  • 3.创建java工程，在项目根目录创建build.xml

    <?xml version="1.0" encoding="UTF-8" ?>
<!-- project name是名字,default是默认的行为,basedir是项目根目录(.代表当前目录) -->
<project name="ant_test" default="run" basedir=".">
    <!-- 自定义的几个目录属性，lib是三方包目录，src是源目录，desc是目标目录，ant_test_jar是打包名 -->
    <property name="lib" value="lib"/>
    <property name="src" value="src"/>
    <property name="dest" value="classes"/>
    <property name="ant_test_jar" value="ant_test.jar"/>
    <!-- 定义三方包路径标识符，有点类似File的作用 -->
    <path id="libPath">
        <fileset dir="${lib}">
            <include name="*.jar" />
        </fileset>
    </path>
    <!-- 给path属性定义一个property -->
    <property name="classpath" refid="libPath"/>

    <!-- 自定义init行为 -->
    <target name="init">
        <!-- 创建classes目录 -->
        <mkdir dir="${dest}"/>
    </target>
    <!-- 自定义编译行为-->
    <target name="compile" depends="init">
        <!-- 编译src目录源代码，目标目录是classes，引用lib中三方jar包 -->
        <javac srcdir="${src}" destdir="${dest}" classpath="${classpath}"/>
    </target>
    <!-- 自定义打包行为-->
    <target name="build" depends="compile">
        <!-- 包名ant_test.jar，根目录classes -->
        <jar jarfile="${ant_test_jar}" basedir="${dest}">
            <!-- 排除测试文件 -->
            <exclude name="**/*Test.*"/>
            <!-- 如果包可执行，需配置manifest的Main-Class -->
            <manifest>
                <attribute name="Main-Class" value="org.example.App"/>
            </manifest>
        </jar>
    </target>
    <!-- 自定义执行工程行为-->
    <target name="run" depends="build">
        <java classname="org.example.App" classpath="${ant_test_jar}"/>
    </target>
    <!-- 自定义清除,删除目录和文件-->
    <target name="clean" depends="run">
        <delete dir="${dest}"/>
        <delete file="${ant_test_jar}"/>
    </target>
    <target name="rerun" depends="clean,run">
        <ant target="clean" />
        <ant target="run" />
    </target>
</project>

  • 生成可执行jar包,执行

    java -jar ant_test.jar
